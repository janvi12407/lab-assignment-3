#include <stdio.h>
int main() {
    int num, rev = 0, digit;
    printf("Enter a number: ");
    scanf("%d", &num);

    // Reverse first to get digits in correct order
    int temp = num;
    while(temp > 0) {
        rev = rev * 10 + temp % 10;
        temp /= 10;
    }
    printf("Digits: ");
    while(rev > 0) {
        digit = rev % 10;
        printf("%d ", digit);
        rev /= 10;
    }
    return 0;
}
