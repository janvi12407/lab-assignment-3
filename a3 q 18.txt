#include <stdio.h>
int main() {
    int i, boys = 0, girls = 0;
    char sex;
    printf("Enter gender of 50 students (M/F):\n");
    for(i = 1; i <= 50; i++) {
        scanf(" %c", &sex);
        if(sex == 'M' || sex == 'm')
            boys++;
        else if(sex == 'F' || sex == 'f')
            girls++;
    }
    printf("Boys = %d, Girls = %d\n", boys, girls);
    return 0;
}
