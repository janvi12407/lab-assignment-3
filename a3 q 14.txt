#include <stdio.h>
int main() {
    int i, n, sumOdd = 0, values[10], sum = 0;
    float mean;

    printf("Enter a number n (for odd sum): ");
    scanf("%d", &n);

    for(i = 1; i <= n; i++) {
        if(i % 2 != 0)
            sumOdd += i;
    }
    printf("Sum of odd numbers up to %d = %d\n", n, sumOdd);

    printf("Enter 10 values:\n");
    for(i = 0; i < 10; i++) {
        scanf("%d", &values[i]);
        sum += values[i];
    }
    mean = sum / 10.0;
    printf("Mean of 10 values = %.2f\n", mean);

    return 0;
}
